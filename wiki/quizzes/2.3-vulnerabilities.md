const progress = root.createDiv({ cls: "quiz-progress" });
let solvedCount = 0;
let wrongCount = 0;
const updateProgress = () => {
  progress.setText(`Progress: ${solvedCount} / ${questions.length} solved · ${wrongCount} wrong attempts`);
};
updateProgress();

questions.forEach((qData, i) => {
  const card = root.createDiv({ cls: "quiz-card" });
  card.createEl("div", { cls: "quiz-q", text: `Q${i + 1}. ${qData.q}` });
  let solved = false;
  const feedbackEl = card.createDiv({ cls: "quiz-feedback", attr: { style: "display:none" } });

  qData.opts.forEach((opt) => {
    const btn = card.createEl("button", { cls: "quiz-opt", text: opt.t });
    btn.onclick = () => {
      if (solved) return;
      if (opt.c) {
        btn.classList.add("correct");
        feedbackEl.style.display = "";
        feedbackEl.className = "quiz-feedback correct";
        feedbackEl.setText("✓ " + opt.w);
        solved = true;
        solvedCount++;
        updateProgress();
      } else {
        btn.classList.add("incorrect");
        btn.disabled = true;
        feedbackEl.style.display = "";
        feedbackEl.className = "quiz-feedback incorrect";
        feedbackEl.setText("✗ " + opt.w);
        wrongCount++;
        updateProgress();
      }
    };
  });
});
