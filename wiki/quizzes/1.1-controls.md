const progress = root.createDiv({ cls: "quiz-progress" });
let solvedCount = 0;
let wrongCount = 0;
const missedQs = new Set();
const updateProgress = () => {
  progress.setText(`Progress: ${solvedCount} / ${questions.length} solved · ${wrongCount} wrong attempts`);
};
updateProgress();

questions.forEach((qData, i) => {
  const card = root.createDiv({ cls: "quiz-card" });
  card.createEl("div", { cls: "quiz-q", text: `Q${i + 1}. ${qData.q}` });
  let solved = false;
  const feedbackEl = card.createDiv({ cls: "quiz-feedback", attr: { style: "display:none" } });

  qData.opts.forEach((opt) => {
    const btn = card.createEl("button", { cls: "quiz-opt", text: opt.t });
    btn.onclick = () => {
      if (solved) return;
      if (opt.c) {
        btn.classList.add("correct");
        feedbackEl.style.display = "";
        feedbackEl.className = "quiz-feedback correct";
        feedbackEl.setText("✓ " + opt.w);
        solved = true;
        solvedCount++;
        updateProgress();
        if (solvedCount === questions.length) persistResults();
      } else {
        btn.classList.add("incorrect");
        btn.disabled = true;
        feedbackEl.style.display = "";
        feedbackEl.className = "quiz-feedback incorrect";
        feedbackEl.setText("✗ " + opt.w);
        wrongCount++;
        missedQs.add(i + 1);
        updateProgress();
      }
    };
  });
});

async function persistResults() {
  try {
    const file = app.workspace.getActiveFile();
    if (!file) return;
    const logPath = file.path.replace(/\.md$/, ".log.md");
    const date = new Date().toISOString().slice(0, 10);
    const missedList = [...missedQs].sort((a, b) => a - b);
    const missedStr = missedList.length === 0 ? "_Clean sweep_" : missedList.map(n => `Q${n}`).join(", ");
    const entry = `### [${date}] Attempt\n- Solved: ${solvedCount}/${questions.length}\n- Wrong attempts: ${wrongCount}\n- Missed (first try): ${missedStr}\n`;
    const exists = await app.vault.adapter.exists(logPath);
    if (exists) {
      const prior = await app.vault.adapter.read(logPath);
      await app.vault.adapter.write(logPath, prior.trimEnd() + "\n\n" + entry);
    } else {
      await app.vault.adapter.write(logPath, `# Attempt log: ${file.basename}\n\n` + entry);
    }
    progress.setText(`Done — ${solvedCount} / ${questions.length} solved · ${wrongCount} wrong · log: ${logPath}`);
  } catch (e) {
    console.error("Persist failed:", e);
    progress.setText(`Done — ${solvedCount} / ${questions.length} solved · ${wrongCount} wrong · log save failed (see console)`);
  }
}
